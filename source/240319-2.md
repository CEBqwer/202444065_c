#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    // 정수 입력받아 사칙연산
    int a, b, result;
    result = 0;

    printf("1. 두 개의 정수 입력: "); 
    scanf_s("%d %d", &a, &b);

    result = a + b;
    printf("2. 덧셈 결과는 %d + %d = %d\n", a, b, result);

    result = a - b;
    printf("3. 뺄셈 결과는 %d - %d = %d\n", a, b, result);

    result = a * b;
    printf("4. 곱셈 결과는 %d * %d = %d\n", a, b, result);
    
    printf("5. 나눗셈 결과는 %d / %d = %f", a, b, (float)a/b);
}
